% Breakable list item: labels hang to the left of the current text block
% via \llap. The parent's \leftskip is saved before \RaggedRight (which
% resets it), then restored and advanced, so nested lists indent additively.
\NewDocumentCommand{\MRFListItem}{m m +m}{%
  \par\addvspace{\MRFListItemSkip}%
  \begingroup
  \parindent=0pt
  \setlength{\parskip}{\MRFListItemSkip}%
  \setlength{\MRFNestedListIndent}{\leftskip}%
  \RaggedRight
  \leftskip = \dimexpr\MRFNestedListIndent+\MRFListLabelWidth+\MRFListLabelGap\relax
  \noindent\llap{\makebox[\MRFListLabelWidth][l]{#2}\hspace*{\MRFListLabelGap}}%
  #3\par
  \endgroup
  \addvspace{\MRFListItemSkip}%
}
